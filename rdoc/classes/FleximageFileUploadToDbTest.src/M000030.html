  def test_should_be_valid_with_image
    p = PhotoDb.new(:image_file => files(:photo))
    assert p.save, 'Record expected to be allowed to save'
    
    p = PhotoDb.find(p.id)
    assert_kind_of Magick::Image, p.load_image
    assert p.image_file_data.size > 0
  end
